pop rax ; get the cur rad

push 0
pop rbx ; counter

push 0
pop rcx ; x 

push 0
pop rdx ; y

call col
scrn
hlt
    col:
        push 0
        pop rcx
        push 40
        push rdx

        ja row
        ret

    row:
        call getRad
        call draw
        
        push rcx
        push 1
        add
        pop rcx

        push rcx
        push 40

        jb row

        push rdx
        push 1
        add
        pop rdx

        jmp col

draw:
    push rax
    push 9.5 ;
    jae back
    push rax
    push 8
    jbe back

    push rcx
    push rdx
    push 40 ;
    mul
    add
    pop rbx
    
    push 1
    pop [rbx] ; rbx
    ret


back:
    ret

getRad:
    push rcx
    push 20
    sub
    push rcx
    push 20
    sub
    mul
    push rdx
    push 20
    sub
    push rdx
    push 20
    sub
    mul
    add
    sqrt
    pop rax
