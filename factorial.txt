in        ; input n                     
pop rax   ; f(n) = n * f(n-1) 

push rax
push 0
jb negativenum

push 1
pop rbx                       

call factorial

return:
    push rbx
    out
    hlt

factorial: 

    push rax
    push 2
    
    jb return

    push rax
    push rbx
    mul
    pop rbx

    push rax
    push 1
    sub
    pop rax

    call factorial
    ret

negativenum:
    push -1
    out 
    hlt
