in        ; input n                     
pop rax   ; f(n) = n * f(n-1) 

push rax
push 0

jb negativenum ; jump if the number is negative                   

call factorial
out
hlt

factorial: 

    push rax
    push 1
    
    jb return

    push rax

    push rax
    push 1
    sub
    pop rax

    call factorial

    mul

    ret

negativenum:
    push -1
    out 
    hlt

return:
    push 1
    ret
